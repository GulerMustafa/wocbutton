## Install

```bash
npm install --save wocbutton
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'wocbutton'
import 'wocbutton/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
